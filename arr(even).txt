#include "stdafx.h"
#include <iostream>
using namespace std;
int main()
{
	int const size = 10;
	int a[size];
	int i, c = 0;
	cout << "input arr"<<endl;
	for (i = 1; i <= size; i++) {
		cin >> a[i];
	}
	for (i = 1; i <= size; i++) {
		for (int j = i + 1; j <= size; j++) {
			if (a[i] == a[j]) {
				a[i] = 0;
				a[j] = 0;
				c+=2;
			}
		}
	}
	cout << endl;
	for (i = 1; i <= size; i++) {
		if (a[i] != 0) {
			cout << a[i] << " ";
		}
	}
	cout << endl<<"even elements : " << c;
	system("pause > nul");
	return 0;
}
